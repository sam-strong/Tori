# File lib/ipaddress/prefix.rb, line 200
def self.parse_netmask(netmask)
  octets = netmask.split(".").map{|i| i.to_i}
  num = octets.pack("C"*octets.size).unpack("B*").first.count "1"
  return self.new(num)
end
