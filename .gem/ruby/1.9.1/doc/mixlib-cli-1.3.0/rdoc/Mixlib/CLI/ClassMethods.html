def banner(bstring=nil)
  if bstring
    @banner = bstring
  else
    @banner ||= "Usage: #{$0} (options)"
    @banner
  end
end

def option(name, args)
  @options ||= {}
  raise(ArgumentError, "Option name must be a symbol") unless name.kind_of?(Symbol)
  @options[name.to_sym] = args
end
