def run_application
  ohai = Ohai::System.new
  if Ohai::Config[:file]
    ohai.from_file(Ohai::Config[:file])
  else
    ohai.all_plugins
  end
  if @attributes.length > 0
    @attributes.each do |a|
      puts ohai.attributes_print(a)
    end
  else
    puts ohai.json_pretty_print
  end
end
