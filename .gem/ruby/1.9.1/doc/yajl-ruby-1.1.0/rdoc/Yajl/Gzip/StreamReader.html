def self.parse(input, options={}, buffer_size=nil, &block)
  if input.is_a?(String)
    input = StringIO.new(input)
  end
  Yajl::Parser.new(options).parse(new(input), buffer_size, &block)
end

def read(len=nil, buffer=nil)
  if val = super(len)
    unless buffer.nil?
      buffer.replace(val)
      return buffer
    end
    super(len)
  else
    nil
  end
end
